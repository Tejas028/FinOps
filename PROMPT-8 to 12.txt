OBJECTIVE
─────────
Build the cost attribution engine that explains WHY cloud spend
changed using SHAP (SHapley Additive exPlanations). It trains a
LightGBM regressor per (cloud × service_category) group on
spend_features, then generates SHAP values that attribute cost
changes to specific drivers (lag spikes, regional shifts, tag
changes, day-of-week effects, etc.).

Outputs are persisted to a new `cost_attributions` table and
consumed by Module 9 (FastAPI) and Module 11 (Dashboard).

Module lives at: attribution/

CREATE TABLE IF NOT EXISTS cost_attributions (
    attribution_date    DATE            NOT NULL,
    cloud_provider      TEXT            NOT NULL,
    service_category    TEXT            NOT NULL,
    account_id          TEXT            NOT NULL,
    environment         TEXT            NOT NULL DEFAULT 'unknown',
    team                TEXT            NOT NULL DEFAULT 'unknown',

    -- Total cost on this date for this group
    total_cost_usd      DOUBLE PRECISION NOT NULL,

    -- Top SHAP drivers (stored as JSONB — key: feature_name, value: shap_value)
    -- Example: {"cost_lag_7d": 42.3, "rolling_mean_30d": -12.1, "is_weekend": 5.2}
    shap_values         JSONB           NOT NULL,

    -- Top 3 drivers by absolute SHAP (for fast API queries)
    top_driver_1        TEXT,           -- feature name
    top_driver_1_value  DOUBLE PRECISION,
    top_driver_2        TEXT,
    top_driver_2_value  DOUBLE PRECISION,
    top_driver_3        TEXT,
    top_driver_3_value  DOUBLE PRECISION,

    -- Model metadata
    model_r2_score      DOUBLE PRECISION,
    feature_count       INTEGER,
    computed_at         TIMESTAMPTZ     NOT NULL DEFAULT NOW(),

    PRIMARY KEY (attribution_date, cloud_provider, service_category,
                 account_id, environment, team)
);

SELECT create_hypertable(
    'cost_attributions', 'attribution_date',
    chunk_time_interval => INTERVAL '30 days',
    if_not_exists => TRUE
);

CREATE INDEX IF NOT EXISTS idx_attr_cloud_date

CREATE INDEX IF NOT EXISTS idx_attr_service_date
    ON cost_attributions (service_category, attribution_date DESC);

CREATE INDEX IF NOT EXISTS idx_attr_top_driver
    ON cost_attributions (top_driver_1, attribution_date DESC);

CREATE INDEX IF NOT EXISTS idx_attr_shap_gin
    ON cost_attributions USING GIN (shap_values);

import lightgbm as lgb
import shap
import pandas as pd
import numpy as np
import joblib, os
from attribution.config import *

class AttributionModel:
    """
    Per-(cloud × service_category) LightGBM regressor.
    Trained on spend_features. SHAP explains each prediction.
    """

    def __init__(self, cloud_provider: str, service_category: str):
        self.cloud_provider    = cloud_provider
        self.service_category  = service_category
        self.model             = None
        self.explainer         = None
        self.r2_score_         = None
        self._model_path       = os.path.join(
            MODEL_REGISTRY_PATH,
            f"{cloud_provider}__{service_category}.joblib"
        )
        self._explainer_path   = os.path.join(
            MODEL_REGISTRY_PATH,
            f"{cloud_provider}__{service_category}__explainer.joblib"
        )

    def fit(self, df: pd.DataFrame) -> dict:
        """
        Input df: rows from spend_features for this group,
                  sorted by usage_date ASC, NaN-filled.

        Steps:
        1. Select SHAP_FEATURE_COLUMNS as X, TARGET_COLUMN as y
        2. Fill remaining NaN: X.fillna(0) — SHAP requires no NaN
        3. Time-ordered split (NO shuffle):
             train: first 70%, val: next 15%, test: last 15%
        4. Train LightGBM with early_stopping on val set:
             lgb.train(LGBM_PARAMS, train_set, valid_sets=[val_set],
                       callbacks=[lgb.early_stopping(20),
                                  lgb.log_evaluation(50)])
        5. Evaluate R² on test set (sklearn.metrics.r2_score)
        6. Build SHAP TreeExplainer:
             self.explainer = shap.TreeExplainer(self.model)
        7. Save model + explainer with joblib
        8. Return: {"r2": float, "best_iteration": int,
                    "train_rows": int, "test_rows": int}
        """

    def explain(self, df: pd.DataFrame) -> pd.DataFrame:
        """
        Compute SHAP values for all rows in df.

        Steps:
        1. Select + fillna SHAP_FEATURE_COLUMNS from df
        2. shap_matrix = self.explainer.shap_values(X)
           → shape: (n_rows, n_features)
        3. Build result DataFrame:
             columns = SHAP_FEATURE_COLUMNS
             index   = df.index
        4. Add column "total_cost_usd" from df for reference
        5. Return DataFrame (one SHAP row per input row)
        """

    def load(self) -> bool:
        """
        Load saved model + explainer from registry.
        Return True if both files exist and load successfully.
        Return False otherwise (triggers retraining in engine).
        """

    def extract_top_drivers(
        self, shap_row: pd.Series, n: int = 3
    ) -> list:
        """
        Given a Series of feature→shap_value for one row,
        return top N by absolute value as:
        [{"feature": str, "value": float}, ...]
        sorted descending by abs(value).
        """

class AttributionRepository:

    def get_features_for_group(
        self,
        cloud_provider: str,
        service_category: str,
        start_date: date,
        end_date: date
    ) -> List[Dict]:
        """
        SELECT feature_date AS usage_date,
               account_id, environment, team,
               total_cost_usd, record_count,
               cost_lag_1d, cost_lag_7d, cost_lag_30d,
               rolling_mean_7d, rolling_std_7d,
               rolling_mean_30d, rolling_std_30d,
               pct_change_1d, pct_change_7d, pct_change_30d,
               z_score_30d,
               day_of_week, day_of_month, week_of_year,
               month, is_weekend, is_month_start, is_month_end
        FROM spend_features
        WHERE cloud_provider = %s
          AND service_category = %s
          AND feature_date BETWEEN %s AND %s
        ORDER BY usage_date ASC
        """

    def get_all_groups(self) -> List[Dict]:
        """
        SELECT DISTINCT cloud_provider, service_category
        FROM spend_features
        ORDER BY cloud_provider, service_category
        """

    def upsert_attributions(
        self, records: List[AttributionRecord]
    ) -> int:
        """
        Bulk upsert into cost_attributions.
        ON CONFLICT (attribution_date, cloud_provider,
                     service_category, account_id,
                     environment, team)
        DO UPDATE SET shap_values=EXCLUDED.shap_values,
                      top_driver_1=EXCLUDED.top_driver_1, ...
                      computed_at=NOW()
        Use execute_values() in batches of 500.
        Serialize shap_values dict → json.dumps() before insert.
        Returns rows upserted.
        """

    def get_attributions(
        self,
        start_date: date,
        end_date: date,
        cloud_provider: Optional[str] = None,
        service_category: Optional[str] = None,
        top_driver: Optional[str] = None
    ) -> List[AttributionRecord]:
        """
        Query cost_attributions with optional filters.
        If top_driver provided: WHERE top_driver_1 = %s
                                   OR top_driver_2 = %s
                                   OR top_driver_3 = %s
        Returns List[AttributionRecord].
        """

from pydantic import BaseModel

class AttributionEngineResult(BaseModel):
    groups_processed:   int
    groups_skipped:     int
    attributions_written: int
    duration_seconds:   float
    avg_r2_score:       float
    errors:             List[str] = []

class AttributionEngine:

    def run(
        self,
        start_date: date,
        end_date: date,
        cloud_provider: Optional[str] = None,
        force_retrain: bool = False
    ) -> AttributionEngineResult:
        """
        For each (cloud × service_category) group:

        1. repo.get_features_for_group(cloud, service, start, end)
        2. If len(rows) < MIN_ROWS_FOR_TRAINING: skip, log
        3. Load or train AttributionModel:
             - model.load() → if False or force_retrain: model.fit(df)
        4. model.explain(df) → shap_df
        5. For each row in shap_df:
             - Build AttributionRecord:
                 attribution_date = row.usage_date
                 shap_values      = {feat: shap_df[feat][i] for feat in SHAP_FEATURE_COLUMNS}
                 top_driver_1/2/3 = model.extract_top_drivers(shap_row, n=3)
                 model_r2_score   = model.r2_score_
                 total_cost_usd   = row.total_cost_usd
        6. repo.upsert_attributions(attribution_records)
        7. Aggregate results into AttributionEngineResult

        Error handling: try/except per group, log to result.errors,
        continue to next group.

        Print rich summary table:
        | Cloud | Service   | Rows | R²    | Top Driver      |
        | aws   | compute   | 365  | 0.87  | rolling_mean_7d |
        | aws   | other     | 365  | 0.72  | cost_lag_1d     |
        """

class PaginationParams:
    """
    Reusable pagination — inject with Depends(PaginationParams).
    Defaults: page=1, page_size=50, max page_size=500.
    """
    def __init__(
        self,
        page:      int = Query(default=1, ge=1),
        page_size: int = Query(default=50, ge=1, le=500)
    ):
        self.page      = page
        self.page_size = page_size
        self.offset    = (page - 1) * page_size

STEP 5 — ROUTER: BILLING
══════════════════════════════════════════════════════════════════

File: api/routers/billing.py

All billing queries run against: normalized_records table
(written by Module 3, read by Modules 5–8, now served here)

─────────────────────────────────────────────────────────────────
GET /billing/summary
─────────────────────────────────────────────────────────────────
Query params:
  start_date:     YYYY-MM-DD (required)
  end_date:       YYYY-MM-DD (required)
  cloud_provider: aws|azure|gcp (optional, repeatable)
  service:        str (optional)
  region:         str (optional)
  + PaginationParams

Response: PaginatedResponse[BillingSummary]

SQL (approximate — implement as parameterized query):
  SELECT
    usage_date::text,
    cloud_provider,
    service,
    region,
    SUM(cost_usd)         AS total_cost_usd,
    COUNT(*)              AS record_count,
    SUM(is_anomaly::int)  AS anomaly_count
  FROM normalized_records
  WHERE usage_date BETWEEN :start AND :end
    AND (:cloud IS NULL OR cloud_provider = ANY(:cloud))
    AND (:service IS NULL OR service = :service)
    AND (:region IS NULL OR region = :region)
  GROUP BY usage_date, cloud_provider, service, region
  ORDER BY usage_date DESC
  LIMIT :limit OFFSET :offset

GET /forecasts/budget-risk
─────────────────────────────────────────────────────────────────
Query params:

  monthly_budget_usd: float (required)
  horizon_days: int (default: 30)

Response:
{
  "cloud_provider":    str,
  "monthly_budget":    float,
  "predicted_total":   float,
  "lower_total":       float,
  "upper_total":       float,
  "breach_risk":       "none"|"possible"|"likely"|"certain",
  "breach_date":       str | null,   # first date upper_bound exceeds budget
  "days_to_breach":    int | null
}

Logic:
  breach_risk =
    "none"     if upper_total < monthly_budget
    "possible" if lower_total < monthly_budget <= predicted_total
    "likely"   if predicted_total < monthly_budget <= upper_total
    "certain"  if lower_total >= monthly_budget
  breach_date = first forecast_date where cumulative upper_bound
                crosses monthly_budget_usd

STEP A3 — ALERT RULES ENGINE
File: alerting/rules.py

Four rule types. Each takes relevant data and returns List[Alert].
All thresholds configurable via alerting/config.py.

File: alerting/config.py
  Z_SCORE_THRESHOLDS = {
      "low":      2.0,
      "medium":   2.5,
      "high":     3.0,
      "critical": 4.0
  }
  SPEND_SPIKE_PCT_THRESHOLD = 50.0   # pct_change_1d > 50% → alert
  BUDGET_WARNING_PCT        = 85.0   # projected >= 85% of budget → alert
  BUDGET_BREACH_PCT         = 100.0  # projected >= 100% → critical alert

class AlertRulesEngine:

    def evaluate_anomaly(
        self, anomaly: dict, billing_context: dict
    ) -> Optional[Alert]:
        """
        Triggered per anomaly row from the anomalies table.
        Map anomaly.severity → alert.severity directly.
        title: "Anomaly: {service} spend spike on {date}"
        message: "Detected {deviation_pct:.1f}% above expected cost
                  (${actual_cost:.2f} vs ${expected_cost:.2f})"
        metadata: { z_score, deviation_pct, detection_method,
                    actual_cost, expected_cost }
        alert_type: "anomaly_detected"
        """

    def evaluate_spend_spike(
        self, feature_row: dict
    ) -> Optional[Alert]:
        """
        Triggered per spend_features row where
        ABS(pct_change_1d) > SPEND_SPIKE_PCT_THRESHOLD.
        severity: "high" if pct_change > 100%, else "medium"
        title: "{cloud} {service} spend up {pct:.0f}% day-over-day"
        alert_type: "spend_spike"
        """

    def evaluate_budget_breach(
        self,
        cloud_provider: str,
        projected_monthly_cost: float,
        monthly_budget_usd: float,
        breach_date: Optional[str]
    ) -> Optional[Alert]:
        """
        pct_used = projected / budget * 100
        If pct_used >= BUDGET_BREACH_PCT:
            severity = "critical"
            title = "{cloud} budget breach projected by {breach_date}"
        elif pct_used >= BUDGET_WARNING_PCT:
            severity = "high"
            title = "{cloud} budget at {pct_used:.0f}% — breach risk"
        Else: return None
        alert_type: "budget_breach_imminent"
        metadata: { projected, budget, pct_used, breach_date }
        """

STEP A4 — ALERT REPOSITORY
File: alerting/repository.py

class AlertRepository:

    def insert_alerts(self, alerts: List[Alert]) -> int:
        """
        Bulk insert into alerts table.
        ON CONFLICT (alert_date, alert_id) DO NOTHING.
        Use execute_values() in batches of 500.
        Returns rows inserted.
        """

    def get_alerts(
        self,
        start_date: date,
        end_date: date,
        severity: Optional[str] = None,
        cloud_provider: Optional[str] = None,
        alert_type: Optional[str] = None,
        is_resolved: Optional[bool] = None,
        limit: int = 100,
        offset: int = 0
    ) -> Tuple[List[Alert], int]:
        """
        Returns (alerts, total_count) for pagination.
        """

    def resolve_alert(self, alert_id: str) -> bool:
        """
        UPDATE alerts SET is_resolved=TRUE, resolved_at=NOW()
        WHERE alert_id = %s
        Returns True if row updated.
        """

    def get_unresolved_count(self) -> Dict[str, int]:
        """
        SELECT severity, COUNT(*) FROM alerts
        WHERE is_resolved = FALSE
        GROUP BY severity
        Returns {"low": N, "medium": N, "high": N, "critical": N}
        """

class AlertingEngine:

    def run(
        self,
        start_date: date,
        end_date: date,
        cloud_provider: Optional[str] = None,
        monthly_budget_usd: Optional[float] = None
    ) -> dict:
        """
        Step 1: Query anomalies table for date range + cloud filter
                → evaluate_anomaly() per row → collect Alert objects

        Step 2: Query spend_features for rows where
                ABS(pct_change_1d) > SPEND_SPIKE_PCT_THRESHOLD
                → evaluate_spend_spike() per row → collect Alerts

        Step 3: If monthly_budget_usd provided:
                Call /forecasts/budget-risk equivalent logic
                (query forecasts table directly — do not call the API)
                → evaluate_budget_breach() → collect Alert if triggered

        Step 4: Deduplicate alerts:
                Skip insert if identical (alert_type, cloud_provider,
                service_category, alert_date) already exists in DB
                (query before bulk insert)

        Step 5: AlertRepository.insert_alerts(all_alerts)

        Step 6: Return summary:
                {
                  "anomaly_alerts": int,
                  "spend_spike_alerts": int,
                  "budget_alerts": int,
                  "total_inserted": int,
                  "total_skipped": int,
                  "duration_seconds": float
                }
        """
